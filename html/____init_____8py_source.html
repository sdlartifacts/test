636 def detect(agent, fill_none=False):
637  """
638  fill_none: if name/version is not detected respective key is still added to the result with value None
639  """
640  result = dict(platform=dict(name=None, version=None))
641  _suggested_detectors = []
642 
643  for info_type in detectorshub:
644  detectors = _suggested_detectors or detectorshub[info_type]
645  for detector in detectors:
646  try:
647  detector.detect(agent, result)
648  except Exception as _err:
649  pass
650 
651  if fill_none:
652  attrs_d = {'name': None, 'version': None}
653  for key in ('os', 'browser'):
654  if key not in result:
655  result[key] = attrs_d
656  else:
657  for k, v in attrs_d.items():
658  result[k] = v
659 
660  return result

663 def simple_detect(agent):

666  """

674  os_list.append(result['os']['name'])

678  (result.get('dist') and result['dist'].get('version')) or (result.get('os') and result['os'].get('version')) or ""

680  browser_version = 'browser' in result and result['browser'].get('version') or ""
681  if browser_version:
