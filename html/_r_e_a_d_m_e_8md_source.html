12 Usage
13 =====
14 
15 ~~~~ {.sourceCode .python}
16 >>> import httpagentparser
17 >>> s = "Mozilla/5.0 (X11; U; Linux i686; en-US) AppleWebKit/532.9 (KHTML, like Gecko) \
18  Chrome/5.0.307.11 Safari/532.9"
19 >>> print httpagentparser.simple_detect(s)
20 ('Linux', 'Chrome 5.0.307.11')
21 >>> print httpagentparser.detect(s)
22 {'os': {'name': 'Linux'},
23  'browser': {'version': '5.0.307.11', 'name': 'Chrome'}}
24 
25 >>> s = "Mozilla/5.0 (Linux; U; Android 2.3.5; en-in; HTC_DesireS_S510e Build/GRJ90) \
26  AppleWebKit/533.1 (KHTML, like Gecko) Version/4.0 Mobile Safari/533.1"
27 >>> print httpagentparser.simple_detect(s)
28 ('Android Linux 2.3.5', 'Safari 4.0')
